75. How do you create a duplicate weather app in JavaScript that fetches weather data from an API?

Hint:
<h2>Use the OpenWeather API to fetch weather data.</h2>
<h3>Example: </h3>
<div> <input type="text" id="cityInput" placeholder="Enter city name">
<button onclick="getWeather()">Get Weather</button>

<div id="weatherResult"></div>

<script>
function getWeather() {
var city = document.getElementById("cityInput").value;
var apiKey = "YOUR_API_KEY"; // You can get this from OpenWeather
var url = `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${apiKey}&units=metric`;

fetch(url)
.then(response => response.json())
.then(data => {
var weatherDescription = data.weather[0].description;
var temperature = data.main.temp;
var humidity = data.main.humidity;

var result = `

Weather in ${city}

${weatherDescription}

Temperature: ${temperature}C

Humidity: ${humidity}%

`;
document.getElementById("weatherResult").innerHTML = result;
})
.catch(error => {
document.getElementById("weatherResult").innerHTML = "

Error fetching weather data. Please try again.

";
});
}
</script>
</div>
