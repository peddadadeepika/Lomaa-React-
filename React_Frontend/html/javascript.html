1. What is a Variable in JavaScript?

Hint:
let name = "John";
// Declaring a variable 'name' and assigning it the string "John"
console.log(name);
// Output: John

2. What are Data Types in JavaScript?

Hint:
let text = "Hello!";
// Declaring a string variable
console.log(typeof text);
// Output: string

3. What is a Function in JavaScript?

Hint:
function greet() {
    console.log("Hello, World!");
}
greet();
// Output: Hello, World!

4. What is an Array in JavaScript?

Hint:
let fruits = ["apple", "banana", "cherry"];
// Declaring an array of fruits
console.log(fruits[0]);
// Output: apple

5. What is an Object in JavaScript?

Hint:
let person = { name: "Alice", age: 25 };
// Declaring an object with properties
console.log(person.name);
// Output: Alice
